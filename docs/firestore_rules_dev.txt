rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    // Verificar si el usuario está autenticado
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Verificar si el usuario está autenticado y es propietario del recurso
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    
    // Verificar si App Check está habilitado correctamente
    function isAppCheckVerified() {
      // MODO DESARROLLO: Temporalmente permitir operaciones sin app check
      // En producción, cambiar a: return request.auth.token.app_check == true;
      return true;
    }
    
    // Permitir acceso a cualquier usuario autenticado (para desarrollo)
    match /{document=**} {
      allow read, write: if isAuthenticated() && isAppCheckVerified();
    }
    
    // Reglas específicas para colecciones principales
    match /shifts/{document} {
      allow read: if isAuthenticated() && isAppCheckVerified();
      allow write: if isAuthenticated() && isAppCheckVerified();
    }
    
    match /users/{userId} {
      allow read: if isAuthenticated() && isAppCheckVerified();
      allow write: if isOwner(userId) && isAppCheckVerified();
      
      match /attendance/{document} {
        allow read: if isAuthenticated() && isAppCheckVerified();
        allow write: if isOwner(userId) && isAppCheckVerified();
      }
    }
    
    match /schools/{schoolId} {
      allow read: if isAuthenticated() && isAppCheckVerified();
      
      match /geofence/{document} {
        allow read: if isAuthenticated() && isAppCheckVerified();
      }
    }
    
    // Colección especial para verificar conectividad
    match /_ping/{document} {
      allow read: if isAuthenticated() && isAppCheckVerified();
    }
  }
}
